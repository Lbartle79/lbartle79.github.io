<script>

// MAP
let map = L.map('map').setView([52.86,-104.60],13)

L.tileLayer(
'https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png',
{maxZoom:19}
).addTo(map)

let points = []
let activeMarkers = []

// 🔥 YOUR DATA WILL GO HERE
points = []

// SORT
points.sort((a,b)=> new Date(a.start) - new Date(b.start))

// TIMELINE
let t=document.getElementById("timeline")

points.forEach(p=>{

let div=document.createElement("div")
div.className="item"

div.innerHTML =
"<b>" + p.start + "</b><br>" +
"End: " + p.end + "<br>" +
"Duration: " + p.duration + "<br>" +
"Lat: " + p.lat + "<br>" +
"Lon: " + p.lon

div.onclick=()=>{

activeMarkers.forEach(m => map.removeLayer(m))
activeMarkers=[]

let marker = L.marker([p.lat,p.lon]).addTo(map)
activeMarkers.push(marker)

map.setView([p.lat,p.lon],16)

}

t.appendChild(div)

})

</script>
